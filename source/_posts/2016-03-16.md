演示的tmux脚本:

``` python    
    
    #!/bin/bash  
    SESSION=$USER  
    COMMAND='http Space http://localhost:8000/sleep'  
      
    tmux new-session -d -s $SESSION  
      
    tmux new-window -t $SESSION -n 'Logs'  
    tmux split-window -h  
    tmux select-pane -t 0  
    tmux send-keys $COMMAND C-m  
    tmux select-pane -t 1  
    tmux send-keys $COMMAND C-m  
    tmux split-window -v  
    tmux send-keys $COMMAND C-m  
    tmux select-pane -t 0  
    tmux split-window -v  
    tmux send-keys $COMMAND C-m  
    # Attach to session  
    tmux attach-session -t $SESSION  
      
```
